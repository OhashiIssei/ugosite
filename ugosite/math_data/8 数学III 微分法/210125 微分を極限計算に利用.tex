x \sin \bunsuu 1x &\cdots\cdots\ \ x \neq 0\ のとき\\

\eeda

\item $L_3=\dlim_{x \to 0} \bunsuu{x^2f(a)-a^2f(x)}{x^2-a^2}$\ \ \ $(a \neq 0)$
